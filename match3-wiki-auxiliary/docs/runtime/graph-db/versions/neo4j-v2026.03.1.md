### 2. Session 管理

```python
def session(

    default_access_mode: str = "WRITE",  # 访问模式: WRITE, READ
) -> Session:

    """
```

### 3. Cypher 查询

```python
def run(
    query: str,                     # Cypher 查询语句
    parameters: dict | None = None, # 查询参数（推荐使用参数化查询）
) -> Result:
    """
    执行 Cypher 查询（Session.run，自动提交事务）
    
    Args:
        query: Cypher 语句，支持参数占位符 $param
        parameters: 参数字典，例如: {"user_id": 123, "name": "Alice"}
    
    Returns:
        Result: 查询结果对象，支持 .data()、.single()、.value() 等方法
    """

def execute_query(
    query: str,                     # Cypher 查询语句
    parameters: dict | None = None, # 查询参数
    routing_: str = "WRITE",        # 路由模式: WRITE, READ
    database_: str | None = None,   # 数据库名称
    result_transformer_: Callable | None = None,  # 结果转换函数
) -> EagerResult:
    """
    执行 Cypher 查询（Driver 级别，推荐方式）
    
    Args:
        query: Cypher 语句
        parameters: 参数字典
        routing_: 路由模式（WRITE 或 READ）
        database_: 数据库名称，None 表示默认数据库
        result_transformer_: 自定义结果转换函数
    
    Returns:
        EagerResult: 包含完整结果的对象
        
    特性:
        - 自动管理会话和事务
        - 自动重试
        - 推荐用于简单查询
    
    Example:
        records, summary, keys = driver.execute_query(
            "MATCH (n:User {id: $id}) RETURN n",
            id=123,
            routing_="READ"
        )
    """
```

### 4. 事务管理

```python
def execute_write(
    transaction_function: Callable[[Transaction], T],  # 事务函数
    **kwargs,                                          # 传递给事务函数的参数
) -> T:
    """
    执行写事务（自动重试）
    
    Args:
        transaction_function: 事务函数，接收 Transaction 对象
        **kwargs: 传递给事务函数的额外参数
    
    Returns:
        T: 事务函数的返回值
        
    特性:
        - 自动管理事务（begin、commit、rollback）
        - 自动重试（网络错误、集群切换等）
        - 推荐用于所有写操作
    """

def execute_read(
    transaction_function: Callable[[Transaction], T],  # 事务函数
    **kwargs,                                          # 传递给事务函数的参数
) -> T:
    """
    执行读事务（自动重试，可路由到副本）
    
    功能同 execute_write，但：
        - 可以路由到只读副本（集群模式）
        - 不允许写操作
    """
```

### 2. Session 和事务管理（推荐模式）

#### 写事务（推荐）

```python
from neo4j import Session

with driver.session() as session:
    def create_game_node(tx, game_name: str, release_date: str):
        query = """
        CREATE (g:Game {
            name: $name,
            release_date: $release_date,
            created_at: datetime()
        })
        RETURN g
        """
        result = tx.run(query, name=game_name, release_date=release_date)
        return result.single()["g"]
    
    # 执行写事务（自动重试）
    game = session.execute_write(
        create_game_node,
        game_name="Honor of Kings",
        release_date="2015-11-26",
    )
    print(f"Created: {game['name']}")
```

#### 读事务

```python
with driver.session() as session:
    def find_games_by_company(tx, company_name: str):
        query = """
        MATCH (c:Company {name: $company})-[:DEVELOPED]->(g:Game)
        RETURN g.name AS game, g.release_date AS date
        ORDER BY g.release_date DESC
        """
        result = tx.run(query, company=company_name)
        return [{"game": record["game"], "date": record["date"]} for record in result]
    
    # 执行读事务（可路由到副本）
    games = session.execute_read(find_games_by_company, company_name="Tencent")
    for game in games:
        print(f"{game['game']} ({game['date']})")
```

**为什么使用 execute_write/execute_read？**
- 自动管理事务生命周期
- 自动重试（网络错误、集群切换）
- 集群模式下自动路由（读操作可到副本）

### 4. 复杂查询

#### 路径查询

```python
with driver.session() as session:
    def find_shortest_path(tx, game1: str, game2: str):
        query = """
        MATCH path = shortestPath(
            (g1:Game {name: $game1})-[*..5]-(g2:Game {name: $game2})
        )
        RETURN path, length(path) AS hops
        """
        result = tx.run(query, game1=game1, game2=game2)
        record = result.single()
        if record:
            return {
                "path": record["path"],
                "hops": record["hops"],

        return None
    
    result = session.execute_read(
        find_shortest_path,
        game1="王者荣耀",
        game2="英雄联盟",
    )
    if result:
        print(f"Found path with {result['hops']} hops")
```

```python
with driver.session() as session:
    def get_company_stats(tx, company_name: str):

            games: games,
            metrics: collect({name: m.name, value: m.value})
```

```python
with driver.session() as session:
    def find_similar_games(tx, game_name: str, top_k: int = 5):
```

```python
with driver.session() as session:
    def batch_create_nodes(tx, nodes: list[dict]):
        query = """
        UNWIND $nodes AS node
        CREATE (n:Entity)
        SET n = node
        RETURN count(n) AS created
        """
        result = tx.run(query, nodes=nodes)
        return result.single()["created"]
    
    # 批量插入 1000 个节点
    batch_size = 1000
    nodes = [
        {"id": i, "name": f"Entity{i}", "type": "Test"}
        for i in range(10000)
    ]
    
    total_created = 0
    for i in range(0, len(nodes), batch_size):
        batch = nodes[i:i+batch_size]
        count = session.execute_write(batch_create_nodes, nodes=batch)
        total_created += count
    
    print(f"Total created: {total_created}")
```

#### 批量创建关系

```python
with driver.session() as session:
    def batch_create_relationships(tx, relationships: list[dict]):
        query = """
        UNWIND $relationships AS rel
        MATCH (a:Entity {id: rel.from_id})
        MATCH (b:Entity {id: rel.to_id})
```

### 6. 图算法

#### PageRank（需要安装 GDS 插件）

```python
with driver.session() as session:
    def run_pagerank(tx, graph_name: str):
        # 1. 创建图投影
        create_graph = """
        CALL gds.graph.project(
            $graph_name,
            'Game',
            'SIMILAR_TO'
        )
        """
        tx.run(create_graph, graph_name=graph_name)
        
        # 2. 运行 PageRank
        pagerank = """
        CALL gds.pageRank.stream($graph_name)
        YIELD nodeId, score
        RETURN gds.util.asNode(nodeId).name AS game, score
        ORDER BY score DESC
        LIMIT 10
        """
        result = tx.run(pagerank, graph_name=graph_name)
        return [{"game": r["game"], "score": r["score"]} for r in result]
    
    top_games = session.execute_read(run_pagerank, graph_name="game_network")
    for game in top_games:
        print(f"{game['game']}: {game['score']:.4f}")
```

## Match3Runtime 接口设计

在 Runtime 中，Neo4j 以 `Driver` 注入：

```python
from typing import Protocol
from neo4j import Driver

@dataclass(frozen=True)
class Match3Runtime:
    config: Config
    env: Env
    logger: Logger
    neo4j: Driver  # Neo4j Driver（连接池）
    # ... 其他依赖
```

**为什么不用 Protocol？**
- `Driver` 是官方标准接口，功能完备
- 无需额外抽象层
- 可以直接替换为兼容的图数据库客户端

**Repository 使用示例**:

```python
class GraphRepository:
    def __init__(self, rt: Match3Runtime):
        self.driver = rt.neo4j
        self.logger = rt.logger
    
    def create_knowledge_graph(
        self,
        entities: list[dict],
        relationships: list[dict],
    ) -> None:
        with self.driver.session() as session:
            # 创建实体
            def _create_entities(tx):
                query = """
                UNWIND $entities AS entity
                MERGE (e:Entity {id: entity.id})
                SET e.name = entity.name,
                    e.type = entity.type,
                    e.metadata = entity.metadata
                """
                tx.run(query, entities=entities)
            
            session.execute_write(_create_entities)
            
            # 创建关系
            def _create_relationships(tx):
                query = """
                UNWIND $rels AS rel
                MATCH (a:Entity {id: rel.from_id})
                MATCH (b:Entity {id: rel.to_id})
                MERGE (a)-[r:RELATED {type: rel.rel_type}]->(b)
                SET r.weight = rel.weight
                """
                tx.run(query, rels=relationships)
            
            session.execute_write(_create_relationships)
    
    def query_related_entities(
        self,
        entity_id: int,
        max_hops: int = 2,
    ) -> list[dict]:
        with self.driver.session() as session:
            def _query(tx):
                query = f"""
                MATCH (e:Entity {{id: $entity_id}})-[*1..{max_hops}]-(related:Entity)
                RETURN DISTINCT related.id AS id, related.name AS name, related.type AS type
                """
                result = tx.run(query, entity_id=entity_id)
                return [dict(r) for r in result]
            
            return session.execute_read(_query)
```

## 性能优化

### 1. 索引创建

```cypher
-- 为常用查询字段创建索引
CREATE INDEX entity_id_idx FOR (e:Entity) ON (e.id);
CREATE INDEX entity_name_idx FOR (e:Entity) ON (e.name);

-- 唯一约束（自动创建索引）
CREATE CONSTRAINT entity_id_unique FOR (e:Entity) REQUIRE e.id IS UNIQUE;

-- 复合索引
CREATE INDEX game_company_idx FOR (g:Game) ON (g.company, g.release_date);

-- 全文索引
CREATE FULLTEXT INDEX game_name_fulltext FOR (g:Game) ON EACH [g.name, g.description];
```

### 2. 查询优化

```cypher
-- ❌ 避免：扫描全表
MATCH (g:Game)
WHERE g.name = 'Honor of Kings'
RETURN g

-- ✅ 推荐：使用索引
MATCH (g:Game {name: 'Honor of Kings'})
RETURN g

-- 使用 EXPLAIN 查看执行计划
EXPLAIN MATCH (g:Game {name: 'Honor of Kings'}) RETURN g

-- 使用 PROFILE 查看实际执行情况
PROFILE MATCH (g:Game)-[:DEVELOPED_BY]->(c:Company) RETURN g, c
```

## 常见问题

### 1. 内存不足

**现象**: `OutOfMemoryError` 或查询缓慢

**解决**:
- 增大堆内存: `NEO4J_server_memory_heap_max__size=4G`
- 增大页缓存: `NEO4J_server_memory_pagecache_size=2G`
- 创建索引减少扫描

### 2. 查询超时

**解决**:
- 使用索引优化查询
- 限制路径深度: `MATCH (a)-[*1..3]-(b)`
- 使用 `LIMIT` 限制结果数量
- 增大 `max_transaction_retry_time`

### 3. 连接失败

**排查**:
- 检查防火墙: `telnet localhost 7687`
- 检查认证: `NEO4J_AUTH=neo4j/password`
- 查看日志: `docker-compose logs neo4j`
